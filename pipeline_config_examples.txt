# Pipeline 配置示例文件
# 
# 这个文件展示了不同场景下的推荐配置
# 复制相应的命令到终端执行即可

# ============================================================
# 场景 1: 标准检测（推荐日常使用）
# - 平衡速度和精度
# - 适用于大多数情况
# ============================================================
python run_inference_pipeline.py \
    --step 3 \
    --threshold 0.5 \
    --fps 25

# ============================================================
# 场景 2: 高精度检测（重要比赛分析）
# - 逐帧检测，不遗漏任何细节
# - 较低的过滤阈值，避免漏检真球
# - 耗时较长但结果最准确
# ============================================================
python run_inference_pipeline.py \
    --step 1 \
    --threshold 0.3 \
    --fps 25

# ============================================================
# 场景 3: 快速预览（初步分析）
# - 更大的检测步长，快速出结果
# - 适合快速查看数据质量
# ============================================================
python run_inference_pipeline.py \
    --step 5 \
    --threshold 0.5 \
    --fps 25

# ============================================================
# 场景 4: 严格过滤（减少误检）
# - 如果发现结果中误检太多
# - 提高阈值可以过滤掉更多假阳性
# ============================================================
python run_inference_pipeline.py \
    --step 3 \
    --threshold 0.7 \
    --fps 25

# ============================================================
# 场景 5: 宽松过滤（避免漏检）
# - 如果发现真球被错误过滤掉
# - 降低阈值可以保留更多检测结果
# ============================================================
python run_inference_pipeline.py \
    --step 3 \
    --threshold 0.3 \
    --fps 25

# ============================================================
# 场景 6: 自定义数据集路径
# - 处理非默认位置的数据集
# ============================================================
python run_inference_pipeline.py \
    --dataset-root "path/to/your/dataset" \
    --output-base "custom_output_dir" \
    --step 3 \
    --threshold 0.5

# ============================================================
# 场景 7: 使用不同的模型权重
# - 测试不同训练版本的模型
# ============================================================
python run_inference_pipeline.py \
    --wasb-weight "pretrained_weights/wasb_tennis_v2.pth.tar" \
    --fp-model "fp_filter/patch_outputs/model_resnet/epoch_50.pth" \
    --step 3 \
    --threshold 0.5

# ============================================================
# 场景 8: 高帧率视频输出
# - 生成更流畅的视频（如果原始数据是高帧率）
# ============================================================
python run_inference_pipeline.py \
    --step 1 \
    --threshold 0.5 \
    --fps 60

# ============================================================
# Windows CMD/PowerShell 用户注意事项
# ============================================================
# 
# 在 Windows 命令行中，多行命令使用 ^ 而不是 \
# 
# 示例（Windows CMD）：
# python run_inference_pipeline.py ^
#     --step 3 ^
#     --threshold 0.5 ^
#     --fps 25
#
# 示例（Windows PowerShell）：
# python run_inference_pipeline.py `
#     --step 3 `
#     --threshold 0.5 `
#     --fps 25
#
# 或者直接一行：
# python run_inference_pipeline.py --step 3 --threshold 0.5 --fps 25

# ============================================================
# 参数调优建议
# ============================================================
#
# --step (检测步长)
#   - 值越小越精确，但处理时间更长
#   - 1: 逐帧检测，最高精度（~5-10分钟/1000帧）
#   - 3: 推荐值，性能与精度平衡（~2-4分钟/1000帧）
#   - 5: 快速模式，适合初步分析（~1-2分钟/1000帧）
#
# --threshold (FP过滤阈值)
#   - 范围：0.0 - 1.0
#   - 0.3: 宽松过滤，保留更多检测（可能包含一些误检）
#   - 0.5: 推荐值，平衡假阳性和真阳性
#   - 0.7: 严格过滤，大幅减少误检（可能丢失一些真球）
#
# --fps (输出视频帧率)
#   - 通常使用原始数据的帧率
#   - 网球视频常见：25fps（PAL）或 30fps（NTSC）
#   - 高速摄影：60fps 或更高
